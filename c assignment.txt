// USING FOR LOOP

#include <stdio.h>
void main() {
    int start, end;
    printf("Enter the starting number: ");
    scanf("%d", &start);
    printf("Enter the ending number: ");
    scanf("%d", &end);
    if (start > end) {
        printf("Starting number should be less than the ending number.\n");
    } else {
        printf("Odd numbers between %d and %d are:\n", start, end);
        for (int number = start; number <= end; number++) {
            if (number % 2 != 0) {
                printf("%d\n", number);
            }
        }
    }
 
}

//USING DO WHILE LOOP

#include <stdio.h>

void main() {
    int start, end;
    printf("Enter the starting number: ");
    scanf("%d", &start);
    printf("Enter the ending number: ");
    scanf("%d", &end);
    if (start > end) {
        printf("Starting number should be less than the ending number.\n");
    } else {
        printf("Odd numbers between %d and %d are:\n", start, end);
        int current_number = start;
        do {
            if (current_number % 2 != 0) {
                printf("%d\n", current_number);
            }
            current_number++;
        } while (current_number <= end);
    }

}

        for (int current = n1 + (n1 % 2); current <= n2; current += 2) {
            printf("%d\n", current);
        }
    } else {
        // Display odd numbers between the digits using a do-while loop
        printf("Odd numbers between %d and %d are:\n", n2, n1);
        int current = n2 + (!(n2 % 2));
        do {
            printf("%d\n", current);
            current += 2;
        } while (current <= n1);
    }
  
}
